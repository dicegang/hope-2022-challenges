\def\QEnd{\QEnd}
\def\BTrue#1#2{#1}
\def\BFalse#1#2{#2}

\def\FNot#1{#1 {\BFalse} {\BTrue}}

% \FDoWithBoolean{action with #1}{boolean}
\def\FDoWithBoolean#1#2{
	#2 {\FExecuteOne{#1}{\BTrue}} {\FExecuteOne{#1}{\BFalse}}

\def\FIfQEnd#1{
	\expandafter\FIfEmpty\expandafter{\FIfQEndA {} #1 {} \! \@ \QEnd \! \! \@}
}
\def\FIfQEndA #1 \QEnd #2 \! #3 \! \@ { #1 #2 }

\def\FChompToQEnd#1\QEnd{}

\newcount\VFModTmp
\def\FMod#1#2{
	\VFModTmp=#1
	\FDiv{\VFModTmp}{#2}
	\FMul{\VFModTmp}{-#2}
	\FAdd{#1}{\VFModTmp}
}


% \FForEach{list}{func with #1}
\def\FForEach#1#2{
	\FForEachA {#2} #1 \QEnd
}
\def\FForEachA#1#2{
	\FIfQEnd{#2} {} {
		\FExecuteOne{#1}{#2}
		\FForEachA{#1}
	}
}
\def\FForEachV#1{
	\expandafter\FForEach\expandafter{#1}
}
\let\FBreak=\FChompToQEnd

% \FListLen{list}{destination register}
\def\FListLen#1#2{
	#2=0
	\FForEach {#1} { \FAdd{#2}{1} }
}
\def\FListLenV#1{
	\expandafter\FListLen\expandafter{#1}
}

% \FIndexSeq{list}{idx}{action taking #1}
\def\FIndexSeq#1#2#3{
	\VFIndexSeqI=#2
	\FForEach {#1} {
		\FIfNum {\VFIndexSeqI}={0} {
			\FExecuteOne{#3}{##1} \FBreak
		} {
			\FAdd{\VFIndexSeqI}{-1}
		}
	}
}
\newcount\VFIndexSeqI
\def\FIndexSeqV#1{
	\expandafter\FIndexSeq\expandafter{#1}
}


\newcount\VRngState
\VRngState="51C3

% Stores a random integer into the register #1
\def\FGetRng#1{
	\FMul{\VRngState}{75}
	\FAdd{\VRngState}{74}
	\FMod{\VRngState}{"10001}
	#1=\VRngState
}


\def\FStoreFlag#1{
	\def\VFlag{}
	\FForEach{#1}{\expandafter\FAppTo\expandafter\VFlag\expandafter{\expandafter{\expandafter`\detokenize{##1}}}}
}

\def\FCheckFlag{
	\FAnd {\FCheckFlagLen} {\FCheckFlagData}
}

\newcount\VFlagLen

\def\FCheckFlagLen{
	\FListLenV \VFlag {\VFlagLen}
	\FIfNum {\VFlagLen}={\CFlagTargetLen}
}

\def\FCheckFlagData{
	\def\VFCheckFlagDataStates{}
	\FForEachV \CFlagTarget {
		\FDoWithBoolean{\FAppTo\VFCheckFlagDataStates{{####1}}}{\FCheckFlagDataA{##1}}
	}
	\def\VFCheckFlagDataResult{\BTrue}
	\FForEachV \VFCheckFlagDataStates {
		##1 {} {\def\VFCheckFlagDataResult{\BFalse}\FBreak}
	}
	\VFCheckFlagDataResult
}
\def\FCheckFlagDataA#1{
	\FGetRng{\VCheckFlagDataIdx}
	\FMod{\VCheckFlagDataIdx}{\VFlagLen}
	\FGetRng{\VCheckFlagDataScratch}
	\FIndexSeqV{\VFlag}{\VCheckFlagDataIdx}{\FAdd{\VCheckFlagDataScratch}{##1}}
	\FMod{\VCheckFlagDataScratch}{"100}
	\FIfNum {\VCheckFlagDataScratch}={#1}
}
\newcount\VCheckFlagDataIdx
\newcount\VCheckFlagDataScratch
